#!/bin/bash
set -e
echo "Запускаем скрипт!"
sleep 7
clear
sleep 7
echo "hostnamectl hostname hq-cli.au-team.irpo"
hostnamectl hostname hq-cli.au-team.irpo
echo "hostnamectl"
hostnamectl
sleep 7
echo "timedatectl set-timezone Europe/Moscow"
timedatectl set-timezone Europe/Moscow
echo "timedatectl"
timedatectl
sleep 7
echo "ip a"
ip a
